plot logerror against year built

data = data.frame(cbind(yearbuilt, logerror))
data_agg = aggregate(data, by=list(data$yearbuilt), FUN=mean)
data_agg = data_agg[,-1]
plot(data_agg$yearbuilt, data_agg$logerror, type = "l")
